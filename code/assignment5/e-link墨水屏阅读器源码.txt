// ==================== 工具函数 ====================

String extractFileName(const String& fullPath) {
    int idx = fullPath.lastIndexOf('/');
    if (idx >= 0) return fullPath.substring(idx + 1);
    return fullPath;
}

// ==================== SPIFFS 管理 ====================

bool initSPIFFS() {
    if (!SPIFFS.begin(true)) {
        Serial.println("SPIFFS Mount Failed");
        return false;
    }
    if (!SPIFFS.exists("/books")) SPIFFS.mkdir("/books");
    return true;
}

String getStorageInfo() {
    size_t total = SPIFFS.totalBytes();
    size_t used = SPIFFS.usedBytes();
    char buf[40];
    sprintf(buf, "%d KB / %d KB", used / 1024, total / 1024);
    return String(buf);
}

// 获取 /books 下第一个文件的名字（用于自动打开）
String getFirstBook() {
    String name = "";
    File root = SPIFFS.open("/books");
    if (root && root.isDirectory()) {
        File file = root.openNextFile();
        if (file && !file.isDirectory()) {
            name = extractFileName(file.name());
            file.close();
        }
        root.close();
    }
    return name;
}

// 书籍列表 API 仍保留，方便网页管理
String listBooks() {
    JsonDocument doc;
    JsonArray arr = doc.to<JsonArray>();
    File root = SPIFFS.open("/books");
    if (root && root.isDirectory()) {
        File file = root.openNextFile();
        while (file) {
            if (!file.isDirectory()) {
                JsonObject b = arr.add<JsonObject>();
                String name = extractFileName(file.name());
                b["name"] = name;
                b["size"] = String(file.size()) + " bytes";
                b["pages"] = 0;
            }
            file.close();
            file = root.openNextFile();
        }
        root.close();
    }
    String out;
    serializeJson(doc, out);
    return out;
}

bool deleteBook(const String& bookName) {
    String path = "/books/" + bookName;
    if (reading.isReading && reading.currentBook == bookName) {
        reading.bookFile.close();
        reading.isReading = false;
        reading.currentBook = "";
        pageIndex.clear();
    }
    return SPIFFS.exists(path) ? SPIFFS.remove(path) : false;
}

bool saveReadingRecord(const String& book, long pos, int page) {
    if (!preferences.begin("reading", false)) return false;
    preferences.putString("bookName", book);
    preferences.putLong("filePos", pos);
    preferences.putInt("page", page);
    preferences.end();
    return true;
}

bool loadReadingRecord(String& book, long& pos, int& page) {
    if (!preferences.begin("reading", true)) return false;
    book = preferences.getString("bookName", "");
    pos = preferences.getLong("filePos", 0);
    page = preferences.getInt("page", 0);
    preferences.end();
    return book.length() > 0;
}

// ==================== 页索引构建（按实际行数切分） ====================

bool buildPageIndex(File& file) {
    pageIndex.clear();
    if (!file || file.size() == 0) return false;
    file.seek(0);

    int lineCount = 0;
    int charInLine = 0;

    while (file.available()) {
        size_t posBeforeChar = file.position();

        uint8_t c = file.read();
        int charLen = 1;
        if ((c & 0x80) == 0x00) charLen = 1;
        else if ((c & 0xE0) == 0xC0) charLen = 2;
        else if ((c & 0xF0) == 0xE0) charLen = 3;
        else if ((c & 0xF8) == 0xF0) charLen = 4;
        else continue;

        for (int i = 1; i < charLen && file.available(); i++) {
            file.read();
        }

        bool isNewline = (charLen == 1 && c == '\n');

        if (isNewline) {
            lineCount++;
            charInLine = 0;
            if (lineCount >= LINES_PER_PAGE) {
                pageIndex.push_back(file.position());
                lineCount = 0;
            }
            continue;
        }

        int charW = (charLen == 1) ? 1 : 2;

        if (charInLine + charW > CHARS_PER_LINE * 2) {
            lineCount++;
            charInLine = 0;
            if (lineCount >= LINES_PER_PAGE) {
                pageIndex.push_back(posBeforeChar);
                lineCount = 0;
                charInLine = charW;   // 当前字符属于新页第一行
                continue;
            }
        }

        charInLine += charW;
    }

    Serial.printf("Page index built: %d pages\n", pageIndex.size() + 1);
    return true;
}

String readPageContent(int page) {
    if (!reading.bookFile || pageIndex.empty() && page != 0) return "";
    size_t start = (page == 0) ? 0 : pageIndex[page - 1];
    size_t end = (page < pageIndex.size()) ? pageIndex[page] : reading.bookFile.size();
    reading.bookFile.seek(start);
    String content;
    while (reading.bookFile.position() < end) {
        content += (char)reading.bookFile.read();
    }
    return content;
}

// ==================== 显示函数 ====================

void displayMessage(const String& title, const String& msg) {
    display.setFullWindow();
    display.firstPage();
    do {
        display.fillScreen(GxEPD_WHITE);
        u8g2.setFontMode(1);
        u8g2.setForegroundColor(GxEPD_BLACK);
        u8g2.setBackgroundColor(GxEPD_WHITE);
        u8g2.setFont(u8g2_font_wqy12_t_gb2312a);

        int titleWidth = u8g2.getUTF8Width(title.c_str());
        u8g2.setCursor((EPD_WIDTH - titleWidth) / 2, 30);
        u8g2.print(title);

        int msgWidth = u8g2.getUTF8Width(msg.c_str());
        u8g2.setCursor((EPD_WIDTH - msgWidth) / 2, 60);
        u8g2.print(msg);
    } while (display.nextPage());
    display.hibernate();
}

void displayText(const String& text, int page, int total) {
    display.setFullWindow();
    display.firstPage();
    do {
        display.fillScreen(GxEPD_WHITE);
        u8g2.setFontMode(1);
        u8g2.setForegroundColor(GxEPD_BLACK);
        u8g2.setBackgroundColor(GxEPD_WHITE);
        u8g2.setFont(u8g2_font_wqy12_t_gb2312a);

        int lineY = 18, lineCount = 0;
        String line;
        int charWidthSum = 0;
        int byteIdx = 0;

        while (byteIdx < text.length() && lineCount < LINES_PER_PAGE) {
            uint8_t c = text[byteIdx];
            int charLen = 1;
            if ((c & 0x80) == 0x00) charLen = 1;
            else if ((c & 0xE0) == 0xC0) charLen = 2;
            else if ((c & 0xF0) == 0xE0) charLen = 3;
            else if ((c & 0xF8) == 0xF0) charLen = 4;
            else { byteIdx++; continue; }

            if (byteIdx + charLen > text.length()) break;
            String oneChar = text.substring(byteIdx, byteIdx + charLen);

            if (oneChar == "\n") {
                u8g2.setCursor(0, lineY);
                u8g2.println(line);
                line = "";
                charWidthSum = 0;
                lineY += 16;
                lineCount++;
                byteIdx += charLen;
                continue;
            }

            int charW = (charLen == 1) ? 1 : 2;

            if (charWidthSum + charW > CHARS_PER_LINE * 2) {
                u8g2.setCursor(0, lineY);
                u8g2.println(line);
                line = "";
                charWidthSum = 0;
                lineY += 16;
                lineCount++;
                if (lineCount >= LINES_PER_PAGE) break;
            }
            line += oneChar;
            charWidthSum += charW;
            byteIdx += charLen;
        }
        if (line.length() > 0 && lineCount < LINES_PER_PAGE) {
            u8g2.setCursor(0, lineY);
            u8g2.println(line);
        }

        // 页码
        u8g2.setFont(u8g2_font_6x10_tf);
        u8g2.setForegroundColor(GxEPD_BLACK);
        char buf[20];
        sprintf(buf, "%d/%d", page, total);
        int pw = u8g2.getUTF8Width(buf);
        u8g2.setCursor(EPD_WIDTH - pw - 2, EPD_HEIGHT - 5);
        u8g2.print(buf);
        u8g2.setFont(u8g2_font_wqy12_t_gb2312a);
    } while (display.nextPage());
    display.hibernate();
}

// ==================== 书籍操作 ====================

bool openBook(const String& bookName) {
    String path = "/books/" + bookName;
    if (!SPIFFS.exists(path)) return false;
    if (reading.bookFile) reading.bookFile.close();
    reading.bookFile = SPIFFS.open(path, "r");
    if (!reading.bookFile) {
        Serial.println("Failed to open file");
        return false;
    }

    reading.currentBook = bookName;
    if (!buildPageIndex(reading.bookFile)) {
        reading.bookFile.close();
        reading.isReading = false;
        Serial.println("Failed to build page index");
        return false;
    }
    reading.totalPages = pageIndex.empty() ? 1 : pageIndex.size() + 1;
    reading.currentPage = 1;
    reading.isReading = true;
    String content = readPageContent(0);
    displayText(content, reading.currentPage, reading.totalPages);
    saveReadingRecord(bookName, 0, 1);
    Serial.printf("Opened book: %s, total pages: %d\n", bookName.c_str(), reading.totalPages);
    return true;
}

bool resumeReading() {
    String bookName; long pos; int page;
    if (!loadReadingRecord(bookName, pos, page)) return false;
    if (!SPIFFS.exists("/books/" + bookName)) return false;
    if (reading.bookFile) reading.bookFile.close();
    reading.bookFile = SPIFFS.open("/books/" + bookName, "r");
    if (!reading.bookFile) return false;
    if (!buildPageIndex(reading.bookFile)) {
        reading.bookFile.close();
        return false;
    }
    reading.currentBook = bookName;
    reading.totalPages = pageIndex.empty() ? 1 : pageIndex.size() + 1;
    reading.currentPage = constrain(page, 1, reading.totalPages);
    reading.isReading = true;
    String content = readPageContent(reading.currentPage - 1);
    displayText(content, reading.currentPage, reading.totalPages);
    Serial.printf("Resumed reading: %s, page %d/%d\n", bookName.c_str(), reading.currentPage, reading.totalPages);
    return true;
}

void nextPage() {
    if (!reading.isReading || reading.currentPage >= reading.totalPages) {
        Serial.println("Cannot go to next page");
        return;
    }
    reading.currentPage++;
    String content = readPageContent(reading.currentPage - 1);
    displayText(content, reading.currentPage, reading.totalPages);
    saveReadingRecord(reading.currentBook, 0, reading.currentPage);
    Serial.printf("Next page: %d/%d\n", reading.currentPage, reading.totalPages);
}

void prevPage() {
    if (!reading.isReading || reading.currentPage <= 1) {
        Serial.println("Cannot go to previous page");
        return;
    }
    reading.currentPage--;
    String content = readPageContent(reading.currentPage - 1);
    displayText(content, reading.currentPage, reading.totalPages);
    saveReadingRecord(reading.currentBook, 0, reading.currentPage);
    Serial.printf("Prev page: %d/%d\n", reading.currentPage, reading.totalPages);
}

// ==================== 按钮处理（简化版） ====================

void onBootSingleClick() {
    Serial.println("Button: single click (next page)");
    if (reading.isReading) nextPage();
}

void onBootDoubleClick() {
    Serial.println("Button: double click (prev page)");
    if (reading.isReading) prevPage();
}

void onBootLongPress() {
    Serial.println("Button: long press");
    if (reading.isReading) {
        // 退出阅读
        reading.bookFile.close();
        reading.isReading = false;
        pageIndex.clear();
        displayMessage("准备就绪", "长按Boot开始阅读");
    } else {
        // 自动打开第一本书
        String first = getFirstBook();
        if (first.length() > 0) {
            openBook(first);
        } else {
            displayMessage("没有书籍", "请通过网页上传");
        }
    }
}

String bodyToString(uint8_t* data, size_t len) {
    char* str = new char[len + 1];
    memcpy(str, data, len);
    str[len] = '\0';
    String result(str);
    delete[] str;
    return result;
}

// ==================== WiFi ====================

void setupWiFi() {
    WiFi.mode(WIFI_STA);
    WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
    int tries = 0;
    while (WiFi.status() != WL_CONNECTED && tries++ < 30) {
        delay(500);
        Serial.print(".");
    }
    if (WiFi.status() == WL_CONNECTED) {
        Serial.println("\nWiFi OK, IP: " + WiFi.localIP().toString());
        MDNS.begin(MDNS_HOSTNAME);
        displayMessage("WiFi已连接", WiFi.localIP().toString());
    } else {
        WiFi.mode(WIFI_AP);
        WiFi.softAP(AP_SSID, AP_PASSWORD);
        Serial.println("AP mode, IP: " + WiFi.softAPIP().toString());
        displayMessage("热点模式", WiFi.softAPIP().toString());
    }
}

// ==================== Setup & Loop ====================

void setup() {
    Serial.begin(115200);
    delay(100);
    display.init(115200, true, 2, false);
    display.setRotation(0);       // 竖屏
    u8g2.begin(display);
    displayMessage("启动中", "正在初始化...");

    if (!initSPIFFS()) {
        displayMessage("错误", "文件系统初始化失败");
        while (1) delay(1000);
    }

    setupWiFi();
    setupAPI();
    setupButton();

    if (!resumeReading()) {
        displayMessage("准备就绪", "长按Boot开始阅读");
    }
}

void loop() {
    handleButton();
    delay(10);
